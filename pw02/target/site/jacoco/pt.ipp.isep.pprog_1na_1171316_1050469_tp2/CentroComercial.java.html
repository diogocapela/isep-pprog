package pt.ipp.isep.pprog_1na_1171316_1050469_tp2;

import java.util.ArrayList;
import java.util.List;

public class CentroComercial {

    private String nome;
    private List<Loja> lojas;

    public CentroComercial(String nome, List<Loja> lojas) {
        this.nome = nome;
        this.lojas = new ArrayList<Loja>();
        for(Loja loja: lojas) {
            if(loja instanceof LojaAncoraExterna) {
                this.lojas.add(new LojaAncoraExterna((LojaAncoraExterna) loja));
            } else if(loja instanceof LojaAncoraPropria) {
                this.lojas.add(new LojaAncoraPropria((LojaAncoraPropria) loja));
            } else if(loja instanceof LojaComumQuiosque) {
                this.lojas.add(new LojaComumQuiosque((LojaComumQuiosque) loja));
            } else if(loja instanceof LojaComumRestauracao) {
                this.lojas.add(new LojaComumRestauracao((LojaComumRestauracao) loja));
            }
        }
    }

    public String getNome() {
        return this.nome;
    }
    
    public List<Loja> getLojas() {
        return this.lojas;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public void setLojas(List<Loja> lojas) {
        this.lojas = lojas;
    }
    
    @Override
    public String toString() {
        return String.format("Nome: %s%n", this.nome);
    }
    
    public void adicionarLoja(Loja loja) {
        if(loja instanceof LojaAncoraExterna) {
            this.lojas.add(new LojaAncoraExterna((LojaAncoraExterna) loja));
        } else if(loja instanceof LojaAncoraPropria) {
            this.lojas.add(new LojaAncoraPropria((LojaAncoraPropria) loja));
        } else if(loja instanceof LojaComumQuiosque) {
            this.lojas.add(new LojaComumQuiosque((LojaComumQuiosque) loja));
        } else if(loja instanceof LojaComumRestauracao) {
            this.lojas.add(new LojaComumRestauracao((LojaComumRestauracao) loja));
        }
    }
    
    public void removerLoja(Loja loja) {
        this.lojas.remove(loja);
    }



    
    
}
